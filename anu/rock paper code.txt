#include <stdio.h>

// Function to add two numbers
int add_numbers(int num1, int num2) {
    return num1 + num2;
}

// Function to subtract two numbers
int subtract_numbers(int num1, int num2) {
    return num1 - num2;
}

int main() {
    int num1, num2;
    int sum_result, subtract_result;

    // Take input from the user
    printf("Enter the first number: ");
    scanf("%d", &num1);

    printf("Enter the second number: ");
    scanf("%d", &num2);

    // Perform addition and subtraction
    sum_result = add_numbers(num1, num2);
    subtract_result = subtract_numbers(num1, num2);

    // Print the results
    printf("The sum of %d and %d is: %d\n", num1, num2, sum_result);
    printf("The difference when %d is subtracted from %d is: %d\n", num2, num1, subtract_result);

    return 0;
}
